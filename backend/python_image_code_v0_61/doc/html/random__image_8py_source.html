00030 # This function generates a grid with random values.
00031 # @param distribution
00032 #        the distribution the values should follow. It should be a response curve generated
00033 #        from the make_distribution_curve function - see the explanation there for details.
00034 # @param dims
00035 #        The dimensions of the grid to return.
00036 # @param period 
00037 #        an integer denoting the size of squares in the grid with the same value. 
00038 #        For instance, if period is set to 2, then the pixels at (0, 0), (0, 1), 
00039 #        (1, 0) and (1, 1) will all have the same value.
00040 def random_grid(distribution, dims, period=1):
00041         grid = Grid2D(dims)
00042 
00043         width, height = dims
00044 
00045         for i in range(0, width, period):
00046                 for j in range(0, height, period):
00047                         r = random()
00048                         value = distribution(r)
00049 
00050                         for index in grid.window_index_iter((i, j), (i + period, j+period)):
00051                                 grid[index] = value

00053         return grid

00055 ## Takes a grid, and returned a slighlty altered version of that grid.
00056 # Each pixel in the new grid is a blend between the corresponding pixel in the old grid
00057 # and another pixel close by. The second pixel is chosen from a square window around the 
00058 # first pixel, with a probability determined by the accumulative probabilities poassed to 
00059 # the function. 
00060 #
00061 # @param grid The grid to process.
00062 #
00063 # @param acc_probs A list of accumulative probabilities. These probabilities are associated 
00064 # with a square. For instance, if list is [0.1, 0.2, 0.3, 0.3, 0.3, 0.4, 0.6, 0.7, 1.0] then 
00065 # the probabilities are:
00066 #
00067 # @code
00068 # 0.1 0.1 0.1
00069 # 0.0 0.0 0.1
00070 # 0.2 0.1 0.3
00071 # @endcode
00072 #
00073 # @param t Is the blend factor.
00074 def corr(grid, acc_probs, t=.9):
00075         width, height = grid.dims
00076         n = int(sqrt(len(acc_probs)))
00077         m = (n - 1) // 2
00078         new_grid = Grid2D(grid.dims, (0, 0, 0, 1))
00079 
00080         for index in grid.index_iter(): 
00081                 r = random()
00082                 k = 0
00083 
00084                 while r > acc_probs[k]:
00085                         k += 1
00086                 
00087                 i, j = index
00088                 x = i + k % n - m
00089                 y = j + k // n - m
00090 
00091                 if x < 0:
00092                         x += width
00093                 elif x >= width:
00094                         x -= width
00095 
00096                 if y < 0:
00097                         y += height
00098                 elif y >= height:
00099                         y -= height
00100 
00101                 new_grid[index] = mix_grey(grid[index], grid[int(x), int(y)], t)
00102 
00103         return new_grid

00105 ## Returns an altered copy of the given grid.
00106 # This is essentially a blur function, with the proportions of blur
00107 # described by a grid. For example, if the grid is 3 by 3, then 9
00108 # pixels are used in to calculate each new pixel. They are blended
00109 # together in the propertions described in the props grid.
00110 # This value is then blended with the center pixel in the ratio described
00111 # by the parameter t. (This is not necessary - it is to make easier comparissons 
00112 # between this function and the corr function).
00113 # @param grid The grid to process
00114 #
00115 # @param props The proportions to blend pixels.
00116 #
00117 # @param t The blend factor to use with the main pixel.
00118 def corr2(grid, props, t = .9):
00119         width, height = grid.dims
00120         n = props.dims[0]
00121         m = (n - 1) // 2
00122         new_grid = Grid2D(grid.dims, (0, 0, 0, 1))
00123 
00124         for index in grid.index_iter():
00125                 new_grey = 0
00126                 for window_index, props_index_iter in zip(grid.wrapped_square_index_iter(index, m), props.index_iter()):

00128                 new_grid[index] = mix_grey(grid[index], new_grey, t)

00150         x, y = center
00151         width, height = grid.dims
